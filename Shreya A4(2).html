# Function to auto-install and load packages
def install_and_load(packages):
    for package in packages:
        try:
            __import__(package)
        except ImportError:
            import subprocess
            subprocess.call(['pip', 'install', package])
            importlib.import_module(package)  # Reload the module after installation
